#6 Enter radius of circle and find diameter, circumference and area

import math
radius = eval(input())
diameter = 2 * radius
circumference = 2 * math.pi * radius
area = math.pi * radius * radius
print(diameter)
print(circumference)
print(area)

#16 Enter marks of five subjects and calculate total, average and percentage

for i in range(5):
    mark = eval(input())
    marks.append(mark)
total = sum(marks)
average = total / 5
percentage = (total / 500) * 100
print(total)
print(average)
print(percentage)

#18 Enter P, T, R and calculate Compound Interest
